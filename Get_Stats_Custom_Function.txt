/**
     * Get custom installs statistics (by John Eberle)
     *
     * Undocumented function long description
     *
     * @param int $hours number of hours back or 0 for all
     * @return {11:return type}
     */
    public function get_stats_custom ($hours = 0)
    {
        $out = array();
        if (! $this->authorized()) {
            $out['error'] = 'Not authorized';
        } else {
            $model = new Managedinstalls_model;
            foreach ($model->get_stats($hours) as $rs) {
                $out[$rs->status."_".$rs->type] = $rs;
            }
        }

        $obj = new View();
        $obj->view('json', array('msg' => $out));
    }
